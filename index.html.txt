const productGrid = document.getElementById("productGrid");
const searchInput = document.getElementById("searchInput");
const categoryFilter = document.getElementById("categoryFilter");

function displayProducts(productsToShow) {
    productGrid.innerHTML = "";
    productsToShow.forEach(product => {
        const card = document.createElement("div");
        card.classList.add("product-card");

        const starPercentage = (product.rating / 5) * 100;

        card.innerHTML = `
            ${product.discount ? `<div class="discount-label">SALE</div>` : ""}
            <img src="${product.image}" alt="${product.name}">
            <div class="product-info">
                <h3>${product.name}</h3>
                <div class="stars-outer">
                    <div class="stars-inner" style="width:${starPercentage}%"></div>
                </div>
                <p>(${product.reviews} Bewertungen)</p>
                <p>${product.description}</p>
                <p class="price">
                    ${product.discount ? `<span class="old-price">€${product.oldPrice.toFixed(2)}</span>` : ""}
                    €${product.price.toFixed(2)}
                </p>
                <a href="${product.link}" target="_blank">Jetzt kaufen</a>
            </div>
        `;
        productGrid.appendChild(card);
    });
}

// Initial anzeigen
displayProducts(products);

// Filter-Event
searchInput.addEventListener("input", filterProducts);
categoryFilter.addEventListener("change", filterProducts);

function filterProducts() {
    const searchTerm = searchInput.value.toLowerCase();
    const selectedCategory = categoryFilter.value;

    const filtered = products.filter(product => {
        const matchesSearch = product.name.toLowerCase().includes(searchTerm);
        const matchesCategory = selectedCategory === "alle" || product.category === selectedCategory;
        return matchesSearch && matchesCategory;
    });

    displayProducts(filtered);
}
